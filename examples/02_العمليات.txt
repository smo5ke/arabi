# ============================================
# 02 - العمليات الحسابية
# ============================================
# هذا الملف يشرح جميع العمليات الحسابية في لغة عربي

س = 10
ص = 3

# --- العمليات الأساسية ---
اطبع("جمع:", س + ص)          # 13
اطبع("طرح:", س - ص)          # 7
اطبع("ضرب:", س * ص)          # 30
اطبع("قسمة:", س / ص)         # 3.333...
اطبع("قسمة صحيحة:", س \ ص)  # 3
اطبع("باقي:", س % ص)         # 1
اطبع("قوة:", س ^ ص)          # 1000

# --- أولوية العمليات ---
اطبع("أولوية:", 2 + 3 * 4)    # 14 (الضرب أولاً)
اطبع("أقواس:", (2 + 3) * 4)  # 20

# --- العمليات على النصوص ---
ص1 = "مرحبا"
ص2 = " بالعالم"
اطبع(ص1 + ص2)              # مرحبا بالعالم
اطبع("مرحبا" * 3)         # مرح�امرحابامرحبا

# --- العمليات على القوائم ---
ق = [1، 2، 3]
اطبع(ق + [4، 5])          # [1, 2, 3, 4, 5]

# --- مقارنات ---
اطبع(5 == 5)    # صح
اطبع(5 != 3)    # صح
اطبع(5 < 10)    # صح
اطبع(5 > 10)    # خطا
اطبع(5 <= 5)    # صح
اطبع(5 >= 6)    # خطا

# --- مقارنات متسلسلة ---
س = 15
اطبع(10 < س < 20)    # صح
اطبع(س > 5 < 20)     # صح

# --- مقارنات الهوية ---
س = 5
اطبع(س يساوي 5)          # صح
اطبع(س يساوي ليس 10)     # صح

# --- العمليات المنطقية ---
اطبع(صح و صح)      # صح
اطبع(صح او خطا)     # صح
اطبع(ليس خطا)        # صح

# --- في / ليس في ---
اطبع(3 في [1، 2، 3])        # صح
اطبع(5 ليس في [1، 2، 3])    # صح
