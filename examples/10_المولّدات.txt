# 10 - المولّدات

دالة عداد(حد):
    س = 0
    بينما س < حد:
        سلم(س)
        س += 1

لكل ع في عداد(5):
    اطبع(ع)

# --- yield from ---
دالة مولّد(مصدر):
    سلم_من(مصدر)

لكل ع في مولّد([1، 2، 3]):
    اطبع(ع)

# --- مولّد فيبوناتشي ---
دالة فيبوناتشي(حد):
    أ، ب = 0، 1
    بينما أ < حد:
        سلم(أ)
        أ، ب = ب، أ + ب

لكل ع في فيبوناتشي(100):
    اطبع(ع)
